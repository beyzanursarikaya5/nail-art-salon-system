document.addEventListener('DOMContentLoaded', () => {

    // 1. DOM ELEMANLARI VE SABİTLER
    let currentStep = 1;
    const urlParams = new URLSearchParams(window.location.search);
    const selectedDesign = urlParams.get('selected_design'); // URL'den
    const selectedDesignId = localStorage.getItem('selected_design'); // Local Storage'dan ID
    const selectedDesignPath = localStorage.getItem('selected_design_path'); // Local Storage'dan Path
    
    // ... (Diğer tüm DOM sabitleri) ...
    const serviceCheckboxes = document.querySelectorAll('input[name="selected_services"]');
    const nailLengthSelect = document.getElementById('nail_length');
    const dateInput = document.getElementById('appointment_date');
    const form = document.getElementById('appointmentForm');

    const nailArtServiceNames = ["Nail Art"];
    const nailArtOptions = document.getElementById('nailArtOptions');
    const customUpload = document.getElementById('customUpload');
    const nailArtCheckbox = Array.from(serviceCheckboxes).find(cb => cb.getAttribute('data-name') === 'Nail Art');
    
    // KRİTİK: Kart detay form inputlarını al
    const cardNumberInput = document.getElementById('card_number');
    const cardExpiryInput = document.getElementById('card_expiry');
    const cardCVCInput = document.getElementById('card_cvc');
    // YENİ: Kart Sahibi Inputu
    const cardHolderInput = document.getElementById('card_holder');


    // ---------------------------------------------------
    // A. WIZARD MANTIĞI (Aşama İlerleme ve Doğrulama)
    // ---------------------------------------------------

    window.showStep = function(step) {
        document.querySelectorAll('.step-content').forEach(el => el.classList.remove('active'));
        document.getElementById(`step-${step}`).classList.add('active');
        document.getElementById('current_step').value = step;

        document.querySelectorAll('.step-dot').forEach(dot => dot.classList.remove('active'));
        document.querySelector(`.step-dot[data-step="${step}"]`).classList.add('active');

        const titles = {
            1: {title: "Randevu Oluşturma Sihirbazı", subtitle: "Adım 1: Hizmet ve Detay Seçimi"},
            2: {title: "Randevu Oluşturma Sihirbazı", subtitle: "Adım 2: Tarih ve Saat Seçimi"},
            3: {title: "Randevu Oluşturma Sihirbazı", subtitle: "Adım 3: Ödeme Yöntemi ve Onay"}
        };
        document.getElementById('wizard-title').textContent = titles[step].title;
        document.getElementById('wizard-subtitle').textContent = titles[step].subtitle;
        
        currentStep = step;
        updateSummaryPanel();
        
        if (step === 3) {
            window.toggleCardDetails();
        }
    }

    function validateStep(step) {
        let isValid = true;
        const stepElement = document.getElementById(`step-${step}`);
        const requiredInputs = stepElement.querySelectorAll('[required]');

        requiredInputs.forEach(input => {
            if (input.style.display !== 'none' && !input.value) { 
                isValid = false;
            }
        });

        if (step === 1 && document.querySelectorAll('input[name="selected_services"]:checked').length === 0) {
            isValid = false;
        }

        if (step === 2 && (!dateInput.value || !document.getElementById('appointment_time_hidden').value)) {
            isValid = false;
        }
        
        // KRİTİK: Adım 3 için kartla ödeme seçiliyse zorunluluk kontrolü (İSİM DAHİL)
        if (step === 3 && document.getElementById('pay_card').checked) {
            if (!cardHolderInput.value || !cardNumberInput.value || !cardExpiryInput.value || !cardCVCInput.value) {
                isValid = false;
            }
        }

        if (!isValid) {
            alert("Lütfen bu adımdaki tüm zorunlu alanları doldurun.");
        }
        return isValid;
    }

    window.nextStep = function(next) {
        if (validateStep(currentStep)) {
            showStep(next);
        }
    }

    window.prevStep = function(prev) {
        showStep(prev);
    }
    
    // ---------------------------------------------------
    // B. FİYAT VE ÖZET MANTIĞI
    // ---------------------------------------------------

    function calculatePrices() {
        let priceServices = 0;
        let priceDesign = 0;
        let priceLength = 0;
        
        serviceCheckboxes.forEach(cb => {
            if(cb.checked) {
                if (cb.getAttribute('data-name') !== 'Nail Art') {
                    priceServices += parseFloat(cb.getAttribute('data-price') || 0);
                }
            }
        });

        if(nailLengthSelect.value === 'ekstra_uzun') {
            priceLength = 100;
        }
        
        const designPriceFromStorage = parseFloat(localStorage.getItem('selected_design_price') || 0);
        const customPriceEl = document.getElementById('custom_design_price_input');
        let customPrice = 0;
        
        if (customUpload.style.display !== 'none' && customPriceEl) {
            customPrice = parseFloat(customPriceEl.value || 0);
        }

        if (designPriceFromStorage > 0) {
            priceDesign = designPriceFromStorage;
        } else if (customPrice > 0) {
            priceDesign = customPrice;
        }

        const charmSelect = document.getElementById('charmSelect');
        let charmPrice = 0;
        if (charmSelect) {
            charmPrice = parseFloat(charmSelect.value) || 0;
        }

        const totalPrice = priceServices + priceDesign + priceLength + charmPrice;

        return {priceServices, priceDesign, priceLength, totalPrice, charmPrice};
    }

    window.updateForm = function() {
        const nailArtChecked = Array.from(serviceCheckboxes).some(c => 
            nailArtServiceNames.includes(c.getAttribute('data-name')) && c.checked
        );

        const requiresNailDetails = ["Protez Tırnak", "Jel Tırnak", "Nail Art"]; 
        const detailRequired = Array.from(serviceCheckboxes).some(c => 
            c.checked && requiresNailDetails.includes(c.getAttribute('data-name'))
        );
        
        const nailLengthEl = document.getElementById('nail_length');
        const nailShapeEl = document.getElementById('nail_shape');
        
        if (detailRequired) {
            nailLengthEl.setAttribute('required', true);
            nailShapeEl.setAttribute('required', true);
        } else {
            nailLengthEl.removeAttribute('required');
            nailShapeEl.removeAttribute('required');
        }

        if (nailArtChecked) {
           nailArtOptions.style.cssText = 'display: flex !important; flex-direction: column; gap: 15px; margin-top: 15px;';
        } else {
           nailArtOptions.style.cssText = 'display: none !important;';
           document.getElementById('customUpload').style.display = 'none';
           localStorage.removeItem('selected_design_price');
           
           const previewEl = document.getElementById('selected_design_preview');
           if (previewEl) {
               previewEl.style.display = 'none';
               previewEl.src = '';
           }
        }

        updateSummaryPanel();
    };
    
    window.updateSummaryPanel = function() {
        const prices = calculatePrices();
        
        document.getElementById('price-services').textContent = prices.priceServices.toFixed(2) + ' TL';
        document.getElementById('price-length').textContent = prices.priceLength.toFixed(2) + ' TL';
        document.getElementById('price-design').textContent = prices.priceDesign.toFixed(2) + ' TL';
        document.getElementById('summary-charm').textContent = prices.charmPrice.toFixed(2) + ' TL';
        document.getElementById('price-final').textContent = prices.totalPrice.toFixed(2) + ' TL';
        
        document.getElementById('final_price_input').value = prices.totalPrice.toFixed(2);
        document.getElementById('final-price-summary').textContent = prices.totalPrice.toFixed(2) + ' TL';

        const selectedServices = Array.from(serviceCheckboxes).filter(cb => cb.checked).map(cb => cb.getAttribute('data-name'));
        document.getElementById('summary-services').textContent = selectedServices.join(', ') || 'Yok';
        
        const lengthText = nailLengthSelect.options[nailLengthSelect.selectedIndex].text;
        document.getElementById('summary-details').textContent = `${lengthText}, Şekil: ${document.getElementById('nail_shape').value || 'Seçilmedi'}`;
        
        const timeVal = document.getElementById('appointment_time_hidden').value;
        const dateVal = dateInput.value;
        document.getElementById('summary-date-time').textContent = (dateVal && timeVal) ? `${dateVal} - ${timeVal}` : 'Seçilmedi';

    }

    // ---------------------------------------------------
    // C. RANDEVU VE ÖDEME MANTIKLARI
    // ---------------------------------------------------

    window.choosePreMade = function() {
        localStorage.removeItem('selected_design');
        localStorage.removeItem('selected_design_price');
        localStorage.removeItem('selected_design_is_nailart');
        window.location.href = '/tasarimlar?return_to=randevu';
    }

    window.uploadCustomDesign = function() {
        localStorage.removeItem('selected_design_price');
        document.getElementById('customUpload').style.display = 'block';
        window.updateForm();
    }
    
    // KRİTİK: Kart Detaylarını Göster/Gizle ve Zorunluluk Ayarı (İSİM DAHİL)
    window.toggleCardDetails = function() {
        const isCard = document.getElementById('pay_card').checked;
        const formEl = document.getElementById('card-details-form');
        
        formEl.style.display = isCard ? 'block' : 'none';

        cardHolderInput.required = isCard;
        cardNumberInput.required = isCard;
        cardExpiryInput.required = isCard;
        cardCVCInput.required = isCard;
        
        if (!isCard) {
            cardHolderInput.value = '';
            cardNumberInput.value = '';
            cardExpiryInput.value = '';
            cardCVCInput.value = '';
        }
    }
    
    form.addEventListener('submit', function(e) {
        e.preventDefault(); 

        if (!validateStep(3)) return; 

        const formData = new FormData(this);
        const actionUrl = this.action;

        fetch(actionUrl, {
            method: 'POST',
            body: formData
        })
        .then(response => {
            if (response.ok) {
                document.getElementById('successModal').style.display = 'flex';
                localStorage.clear(); 
            } else {
                response.json().then(data => {
                    alert('Hata: ' + (data.message || 'Randevu oluşturulamadı.'));
                });
            }
        })
        .catch(error => {
            console.error('Submit Hatası:', error);
            alert('Ağ hatası veya sunucu bağlantı hatası oluştu.');
        });
    });

    // ---------------------------------------------------
    // D. SAATLER VE BAŞLANGIÇ
    // ---------------------------------------------------

    window.dateChangeHandler = function() {
        const selectedDate = dateInput.value;
        const timeBlocksContainer = document.getElementById('time-blocks');
        timeBlocksContainer.innerHTML = '';
        if (!selectedDate) return;

        fetch(`/api/booked-hours?date=${selectedDate}`) 
            .then(res => res.json())
            .then(bookedHours => {
                const startHour = 9;
                const endHour = 18;
                const interval = 2; 
                document.getElementById('appointment_time_hidden').value = ''; 

                for(let h = startHour; h <= endHour; h += interval){
                    let hourStr = h.toString().padStart(2,'0') + ":00";
                    const block = document.createElement('button');
                    block.type = 'button';
                    block.textContent = hourStr;
                    block.className = 'btn-nav';

                    if(bookedHours.includes(hourStr)){
                        block.disabled = true;
                        block.style.cssText = 'background: #ffdcdc; color: #888; cursor: not-allowed;';
                    } else {
                         block.style.cssText = 'background: #fff0fa; color: #ff3399; border: 1px solid #ffccea;';
                    }

                    block.addEventListener('click', () => {
                        if(block.disabled) return; 
                        document.querySelectorAll('#time-blocks button').forEach(b => {
                            if(!b.disabled) b.style.cssText = 'background: #fff0fa; color: #ff3399; border: 1px solid #ffccea;';
                        });
                        
                        block.style.cssText = 'background: #ff3399; color: white; border: 1px solid #ff3399;';
                        document.getElementById('appointment_time_hidden').value = hourStr;
                        updateSummaryPanel();
                    });
                    timeBlocksContainer.appendChild(block);
                }
                updateSummaryPanel();
            })
            .catch(error => {
                console.error('Fetch hatası:', error);
                timeBlocksContainer.innerHTML = '<p style="color:red;">Saatler yüklenemedi.</p>';
            });
    }

    // --- BAŞLANGIÇ YÜKLEME ---

            previewEl.style.display = 'block';

def allowed_file(filename):
    return '.' in filename and \
           filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS

def is_over_18(dob_str):
    today = datetime.today()
    try:
        birth_date = datetime.strptime(dob_str, '%Y-%m-%d')
    except ValueError:
        return False 
    age = today.year - birth_date.year - ((today.month, today.day) < (birth_date.month, birth_date.day))
    return age >= 18

def login_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'logged_in' not in session or not session['logged_in']:
            flash('Bu sayfaya erişim için giriş yapmalısınız.', 'error')
            return redirect(url_for('auth_page', next=request.path))
        return f(*args, **kwargs)
    return decorated_function

@app.route('/login', methods=['POST'])
def login():
    if request.method == 'POST':
        email = request.form.get('email')
        password_raw = request.form.get('password')
        next_url = request.args.get('next') 

        cur = mysql.connection.cursor()
        cur.execute("SELECT user_id, username, password_hash, full_name, is_admin, profile_image FROM users WHERE username = %s", (email,))
        user = cur.fetchone()
        cur.close()

        if user and check_password_hash(user['password_hash'], password_raw):
            session['logged_in'] = True
            session['user_id'] = user['user_id']
            session['full_name'] = user['full_name']

            session['profile_image'] = user.get('profile_image')

            flash('Giriş başarılı!', 'success')

            # YÖNLENDİRME MANTIĞI
            if session['is_admin']:
                return redirect(url_for('admin_dashboard'))
            elif next_url:
                return redirect(next_url)
            else:
                return redirect(url_for('profile_page'))
        else:
            return redirect(url_for('auth_page', error='E-posta veya şifre hatalı.', next=next_url))
            
    return redirect(url_for('auth_page'))

        try:
            cur = mysql.connection.cursor()

            if cur.fetchone():
                cur.close()
                return redirect(url_for('auth_page', error='Bu e-posta zaten kayıtlı.'))

            sql = """INSERT INTO users (username, password_hash, full_name, profile_completed, is_admin)

            cur.execute(sql, (email, password_hashed, full_name))
            mysql.connection.commit()
            new_user_id = cur.lastrowid
            cur.close()
            
            session['logged_in'] = True
            session['user_id'] = new_user_id
            session['full_name'] = full_name 
            session['is_admin'] = False # Yeni kayıt yönetici değildir
            
            return redirect(url_for('profile_setup_page'))
        
        except Exception as e:
            print(f"SQL HATA: {e}")

def profile_page(user_id=None):
    # 1. YÖNETİCİ KONTROLÜ: Eğer yönetici profil sayfasına girmek isterse, Dashboard'a at.
    if session.get('is_admin'):
        return redirect(url_for('admin_dashboard'))

    current_user_id = session.get('user_id')
    # Başkasının profiline girmeyi engelle (Opsiyonel)
    target_id = user_id if user_id else current_user_id
    if target_id != current_user_id:
         target_id = current_user_id

    cur = mysql.connection.cursor()
    
    # Kullanıcı bilgisi
    cur.execute("SELECT * FROM users WHERE user_id = %s", [target_id])
    user_info = cur.fetchone()
    
    # Randevuları Çek
    cur.execute("""
        SELECT a.*, d.design_name, d.design_image, d.design_id
        FROM appointments a
        LEFT JOIN designs d ON a.design_id = d.design_id
        WHERE a.user_id = %s
        ORDER BY a.appointment_date DESC
    """, [target_id])
    
    all_appointments_base = cur.fetchall()
    
    # Hizmet İsimlerini Çek
    cur.execute("SELECT service_id, name FROM services")
    services_dict = {row['service_id']: row['name'] for row in cur.fetchall()}
    
    future_appointments = []
    past_appointments = []
    now = datetime.now()
    
    for appt in all_appointments_base:
        # Randevuya ait hizmetleri bul
        cur.execute("SELECT service_id FROM appointment_services WHERE appointment_id = %s", [appt['appointment_id']])
        s_ids = [r['service_id'] for r in cur.fetchall()]
        appt['service_name'] = ', '.join([services_dict.get(sid, '') for sid in s_ids])
        
        # Özel tasarım varsa görsel yolunu ayarla
        if appt.get('custom_design_path'):
            appt['appointment_image_path'] = appt['custom_design_path']
        else:
            appt['appointment_image_path'] = None

        if appt['appointment_date'] > now:
            future_appointments.append(appt)
        else:
            past_appointments.append(appt)
    
    cur.close()
    return render_template('profile.html', user_info=user_info, future_appointments=future_appointments, past_appointments=past_appointments)

        GROUP BY a.appointment_id
        ORDER BY a.appointment_date DESC
    """)

        new_total_price = request.form.get('total_price') 
        new_length = request.form.get('nail_length')
        new_shape = request.form.get('nail_shape')
        
        new_datetime = datetime.strptime(f"{new_date} {new_time}", '%Y-%m-%d %H:%M')
        new_design_id = int(new_design_id_raw) if new_design_id_raw and new_design_id_raw.isdigit() else None
        
        try:
            # Ana Randevuyu Güncelle
            cur.execute("""
                UPDATE appointments SET 
                    appointment_date = %s, 
                    design_id = %s,
                    total_price = %s,
                    nail_length = %s,  
                    nail_shape = %s,
                    status = 'pending' 
                WHERE appointment_id = %s AND user_id = %s
            """, (new_datetime, new_design_id or None, new_total_price, new_length, new_shape, appointment_id, user_id))
            
            # Hizmetleri Sıfırla ve Yeniden Ekle
            cur.execute("DELETE FROM appointment_services WHERE appointment_id = %s", [appointment_id])
            for service_id_str in new_service_ids:
                cur.execute("""
                    INSERT INTO appointment_services (appointment_id, service_id) VALUES (%s, %s)
                """, (appointment_id, int(service_id_str)))

            mysql.connection.commit()
            flash('Randevu güncellendi.', 'success')
            cur.close()
            return redirect(url_for('profile_page'))

        except Exception as e:
            mysql.connection.rollback()
            flash(f'Hata: {e}', 'error')
            cur.close()
            return redirect(url_for('update_appointment', appointment_id=appointment_id))

    # --- GET (Formu Gösterme) ---
    cur.execute("SELECT service_id FROM appointment_services WHERE appointment_id = %s", [appointment_id])
    selected_service_ids = [row['service_id'] for row in cur.fetchall()]
    
    cur.execute("SELECT * FROM services")
    services = cur.fetchall()
    cur.execute("SELECT * FROM designs")
    designs = cur.fetchall()
    
    cur.close()
    
    return render_template('randevu.html', 
                           services=services, 
                           designs=designs, 
                           today=datetime.now(), 
                           is_update=True, 
                           appointment=appt, 
                           initial_date=appt['appointment_date'].strftime('%Y-%m-%d'), 
                           initial_time=appt['appointment_date'].strftime('%H:%M'),
                           selected_service_ids=selected_service_ids,
                           appointment_id=appointment_id)

@app.route('/delete_appointment/<int:appointment_id>', methods=['POST'])
@login_required
def delete_appointment(appointment_id):
    user_id = session.get('user_id')
    cur = mysql.connection.cursor()
    cur.execute("DELETE FROM appointments WHERE appointment_id=%s AND user_id=%s", (appointment_id, user_id))
    mysql.connection.commit()
    cur.close()
    flash('Randevu silindi.', 'success')
    return redirect(url_for('profile_page'))

@app.route('/api/booked-hours')
def get_booked_hours():
    date_str = request.args.get('date')
    if not date_str: return jsonify([])
    
    cur = mysql.connection.cursor()
    cur.execute("SELECT DATE_FORMAT(appointment_date, '%%H:%%i') as time FROM appointments WHERE DATE(appointment_date) = %s", [date_str])
    times = [r['time'] for r in cur.fetchall()]
    cur.close()
    return jsonify(times)
